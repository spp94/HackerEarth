import java.util.Scanner;
 
class TestClass {
    public static void main(String args[] ) throws Exception {
        Scanner b=new Scanner(System.in);
        int t;
        t=b.nextInt();
        while(t>0)
        {
        	int p;
        	p=b.nextInt();
        	if(p==1)
        	System.out.println(p+" 0");
        	else
        	System.out.println((p-1)+" 1");
        	        	t--;
        }
        
    }
}
