#include <stdio.h>
void fact(int n)
{
	int a[10],p=0;
	int f=1,i;
	for(i=1;i<=n;i++)
	{
		f=f*i;
	}
	int c=0,m=f;
	do
	{
		f=f/10;
		++c;
	}while(f!=0);
	if(c<=5)
	{
	int y=5-c;
	for(i=1;i<=y;i++)
	printf("0");
	printf("%d\n",m);
   }
   else
   {
   do
   {
   	int r=m%10;
   	if(r!=0)
   	a[p]=r;
   	else
   	a[p]=0;
   	++p;
   	m=m/10;
   }while(m!=0);
   	int j;
   	for(j=4;j>=0;j--)
   	printf("%d",a[j]);
   }
 
}
 
int main()
{
    int T,i;
    scanf("%d",&T);
    for(i=0;i<T;i++)
    {
    	int n;
    	scanf("%d\n",&n);
    	fact(n);
    }
}
